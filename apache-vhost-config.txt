# Пример конфигурации виртуального хоста Apache для проекта d:/АН ФАТТОРИЯ

<VirtualHost *:80>
    ServerName localhost
    DocumentRoot "d:/АН ФАТТОРИЯ"

    <Directory "d:/АН ФАТТОРИЯ">
        Options Indexes FollowSymLinks
        AllowOverride All
        Require all granted
    </Directory>

    ErrorLog "logs/AN_Fattoria_error.log"
    CustomLog "logs/AN_Fattoria_access.log" common
</VirtualHost>

# Инструкции:
# 1. Скопируйте этот блок в файл конфигурации Apache, например, в httpd-vhosts.conf
# 2. Убедитесь, что в httpd.conf подключен файл виртуальных хостов (Include conf/extra/httpd-vhosts.conf)
# 3. Перезапустите Apache
# 4. Откройте в браузере http://localhost/наши%20обьекты.html
